// meta caractere tem funções especiais
// aparentemente no JS as barras são / ao invés de \
// cada linguagem tem suas peculiaridades em relação a regex


//Quantifier
. - qualquer caractere
? - zero ou uma vez
+ - uma ou mais vezes 
* - zero ou mais vezes
^ - nega caractere seguinte
  [^,]+ - qualquer dígito, que NÃO vírgula
{n} - exatamente N vezes 
{n,} - no mínimo N vezes 
{n,m} - no mínimo N + 1 vezes, no máximo M vezes;

Classes de char []
[a-z][A-Z][0-9]
\d = numeros
\D = tudo menos numeros
\w - word char = [A-Za-z0-9_]
\W = oposto tudo menos = [^A-Za-z0-9_]
\s - whitspace = [ \t\r\n\f]
\S - seleciona tudo que não espaço

Ancoras
\b - word boundary - delimita o fim da palavra
^ - tb é meta char
$ - tb é meta char

/i - ignore sensitive case
/g - vai retornar todas patterns encontradas, não apenas uma;
  ex. [Rep, Rep, Rep]
  /Rep/ = Rep
  /Rep/g = Rep, Rep, Rep

condição look ahead
(?=u) = vai selecionar o que antecede se na frente tiver "u"
    imprime q: "quit=qu"  /q(?=u)/
(?!u) = vai selecionar o que antecede se lookahead não tive "u"
    imprime q: "quit=qt"  /q(?!u)/
